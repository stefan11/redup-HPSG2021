\section{Introduction}

\frame{
\frametitle{Verbal reduplication in Mandarin Chinese}

\begin{itemize}
\item<1-> Mandarin Chinese: reduplication of verbs expresses a delimitative aspectual meaning \citep[e.g.][]{Chao1968, Chen2001, Dai1997, Li1996, LiThompson1981, Tsao2001, XiaoMcEnery2004, Yang2003, Zhu1998}.
\item<2-> Delimitativeness:\\
``A short duration (i.e. transitoriness) and/or a low iteration frequency”\\ \citep[155]{XiaoMcEnery2004}
  
\ea\label{ex:redup-ex} 
	\ea
	\gll qing ni chang zhe dao cai.\\
	please you taste this \textsc{clf} dish\\
	\glt `Please taste this dish.'
	
	\ex
	\gll qing ni \alert{chang-chang} zhe dao cai.\\
	please you taste-taste this \textsc{clf} dish\\
	\glt `Please taste this dish a little bit.' 
	\z
\z

\item<3-> We aim to provide a formal and unified analysis for the structure of verbal reduplication in Mandarin Chinese.

\end{itemize}
}

\subsection{Forms}

\frame{
\frametitle{Forms}

\ea for monosyllabic verbs: \emph{shuo} `say'
		\ea \gll shuo-shuo\\
		say-say\\\jambox{AA}
\pause
		\ex \gll shuo-yi-shuo\\
		say-one-say\\\jambox{A-\emph{yi}-A}
\pause
		\ex \gll shuo-le-shuo\\
		say-\textsc{pfv}-say\\\jambox{A-\emph{le}-A}
\pause
		\ex \gll shuo-le-yi-shuo\\
		say-\textsc{pfv}-one-say\\\jambox{A-\emph{le}-\emph{yi}-A}
\pause
		\ex \gll shuo-shuo-kan\\
		say-say-look\\\jambox{AA-\emph{kan}}
\pause
		\ex \gll shuo-kan-kan\\
		 say-look-look\\\jambox{A-\emph{kan}-\emph{kan}}
\z
\z
}

\frame{
\frametitle{Forms}

\ea for disyllabic verbs: \emph{lai-wang} come-go `come and go/communicate'\\
		\ea \gll lai-wang-lai-wang\\
		come-go-come-go\\ \jambox{ABAB}
\pause
		\ex \gll lai-wang-le-lai-wang\\
		come-go-\textsc{pfv}-come-go\\ \jambox{AB-\emph{le}-AB}
\pause
		\ex \gll lai-lai-wang-wang\\
		come-come-go-go\\ \jambox{AABB}
\pause
		\z
\z
}

\frame{
\frametitle{Forms}

\ea for V-O compounds: \emph{chang-ge} sing-song `sing'\\
		\ea \gll chang-chang-ge\\
		sing-sing-song\\ \jambox{AAB}
\pause
		\ex \gll chang-yi-chang-ge\\
		sing-one-sing-song\\ \jambox{A-\emph{yi}-AB}
\pause
		\ex \gll chang-le-chang-ge\\
		sing-\textsc{pfv}-sing-song\\ \jambox{A-\emph{le}-AB}
\pause
		\z
\z
}

\frame{
\frametitle{Forms}

\begin{itemize}
    \item There seems to be a fundamental difference between AA, ABAB and AABB \citep{Arcodiaetal2014, Fan1964, MelloniBasciano2018, Xie2020}.
    \item We focus on the AA, A-\emph{yi}-A, A-\emph{le}-A, A-\emph{le}-\emph{yi}-A and ABAB forms only.
\end{itemize}
}


%%%%%%%%%%%%%%% distribution
\subsection{Syntactic distribution}

\frame{
\frametitle{Syntactic distribution}

\begin{itemize}
\item similar to an unreduplicated verb
\pause
\item Cannot combine with an expression of quantity.
\ea
    \ea[]{\gll ta yi tian pao shi li.\\
		he one day run ten mile\\
		\glt `He runs ten miles a day.'}
	\ex[*]{\gll ta yi tian \alert{pao-pao} shi li.\\
	    he one day run-run ten mile\\}
    \z
\z
\pause
Probably because the reduplication already contains a quantity meaning\\\citetext{\citealp[114--115]{Chen2005}; \citealp[84]{Li1998}}.

\pause
\item Cannot combine with aspect markers other than \emph{le}.
\end{itemize}
}



%%%%%%%%%%%%%%%%%%% sem
\subsection{Semantics}

\frame{
\frametitle{Semantics}

\begin{itemize}[<+->]
    \item Core meaning: delimitativeness
    \item A-\emph{yi}-A: same core meaning as AA \citep{Fan1964, Yang2003}
    \item A-\emph{le}-A: a hierarchical combination of perfective and delimitativeness\\ \citep[151]{XiaoMcEnery2004}
\end{itemize}
}

\frame{
\frametitle{Interaction with aspect markers}
\begin{itemize}
    \item The incompatibility with aspect markers other than \emph{le} is due to their semantics.
\pause
    \item Reduplication: holistic and dynamic viewpoint\\\citep{Dai1997, XiaoMcEnery2004}
    \begin{itemize}
        \item Holisticity: the situation as a non-decomposable whole
\pause
        \item Dynamicity: a process full of change
\pause
        \eal
  \ex\label{ex:le-dyn-look}
    \gll Wu Xumang kan-le zuo-an shi liuxia de jiaoyin\\
    Wu Xumang look-\textsc{pfv} commit-crime when leave \textsc{de} footprint\\
    \glt `Wu Xumang looked at the footprint left when the crime was committed.'
  \ex\label{ex:redup-dyn-look}
    \gll Wu Xumang \alert{kan-le-kan} zuo-an shi liuxia de jiaoyin\\
    Wu Xumang look-\textsc{pfv}-look commit-crime when leave \textsc{de} footprint\\
    \glt `Wu Xumang looked a little bit at the footprint left when the crime was committed.'
  \zl
    \end{itemize}
\end{itemize}
}

\frame{
\frametitle{Interaction with aspect markers}
\begin{itemize}[<+->]
    \item \emph{zhe} `\textsc{dur}', \emph{zai} `\textsc{prog}': imperfective aspects \citep{XiaoMcEnery2004}
    \item \emph{guo} `\textsc{exp}': a change out of a state \citep{XiaoMcEnery2004}
    \item \emph{le}: compatible with both changing processes and changing points\\ \citep{XiaoMcEnery2004}
\ea
  \ea\label{ex:guo-dyn}
    \gll ta dang-guo bing.\\
    he serve.as-\textsc{exp} soldier\\
    \glt `He once served as a soldier.'
  \ex\label{ex:le-dyn}
    \gll ta dang-le bing.\\
    he serve.as-\textsc{pfv} soldier\\
    \glt `He became a soldier.'
  \ex
    \gll Weici, Deng Lijun shangxin de ku-le san tian.\\
    for.this Deng Lijun sadly \textsc{de} cry-\textsc{pfv} three day\\
    \glt `For this reason, Deng Lijun cried sadly for three days.'
  \z
\z
\end{itemize}
}

\frame{
\frametitle{Previous analyses}
\begin{itemize}[<+->]
    \item The reduplicant as a verbal classifier \citep{Chao1968, Fan1964, Xiong2016}
    \begin{itemize}
        \item[+] Similar function of constraining the event
        \item[$-$] Differences in syntactic behaviors
    \end{itemize}
    \item The reduplicant as an aspect marker \citep{Arcodiaetal2014, BascianoMelloni2017, Travis1999, Travis2000, YangWei2017}
    \begin{itemize}
        \item[+] Similar function of conveying aspect
        \item[$-$] Difficulties with A-\emph{yi}-A
    \end{itemize}
    \item A special structure for reduplication\\ \citep{FanSongBond2015, Ghomeshietal2004, Travis2001, Travis2003}
    \begin{itemize}
        \item[+] No problem with A-\emph{yi}-A
        \item[$-$] The connection with aspect markers is not captured.
    \end{itemize}
\end{itemize}

\frame{
\frametitle{Previous HPSG analysis}

\begin{itemize}
    \item Reduplication is modeled as a lexical rule.
\end{itemize}

\ea\label{avm:fsb-redup}
\avm{
[\type*{redup-type}\\
cat|head & \1\\
val & \2\\
cont & \3 \normalfont \textsc{hook} [ltop & \4\\
			ind & \5]\\
c-cont & <[\type*{event-rel}\\
		pred & intensifier\_x\_rel\\
		lbl & \4\\
		arg1 & \5]>
]
}
$\mapsto$
\avm{
[cat|head & \1\\
val & \2\\
cont & \3
]
}
\z
}

\frame{
\frametitle{Previous HPSG analysis}

\ea\label{avm:fsb-redup-a}
\avm{
[\type*{redup-a-lr $\subset$ redup-type}\\
cat|head & adjective\\
val & [spr <>]\\
c-cont & <[pred & redup\_up\_x\_rel]>
]
}\\
\textsc{orthography}: A $\to$ AA; (irregular AB $\to$ AABB)
\z

\ea\label{avm:fsb-redup-v}
\avm{
[\type*{redup-v-lr $\subset$ redup-type}\\
cat|head & verb\\
cont|hook & [aspect & non-aspect]\\
c-cont & <[pred & redup\_down\_x\_rel]>
]
}\\
\textsc{orthography}: A $\to$ AA; A $\to$ A-\emph{yi}-A; (irregular AB $\to$ ABAB)
\z
}

\frame{
\frametitle{Previous HPSG analysis}

\begin{itemize}[<+->]
    \item Advantages: 
    \begin{itemize}[<+->]
        \item a unified analysis for verbal and adjectival reduplication
        \item A-\emph{yi}-A as an alternative form of AA
    \end{itemize}
    \item Problems: 
    \begin{itemize}[<+->]
        \item A-\emph{le}-A is not possible because aspect markers are all blocked.
        \item ABAB and AAB are handled as irregular forms even though they are productive \citep{BascianoMelloni2017, MelloniBasciano2018, Xie2020, Xing2000stat}.
    \end{itemize}
\end{itemize}
}

\frame{
\frametitle{Advantages of the current analysis}

\begin{itemize}[<+->]
    \item a unified account for all forms of verbal reduplication in Mandarin Chinese (broader coverage than \citet{FanSongBond2015})
    \item A-\emph{yi}-A as an alternative form of AA
    \item The form and the semantics of A-\emph{le}-A are correctly captured.
    \item compatible with disyllabic verbs
    \item All productive forms are derivable from lexical rules.
\end{itemize}
}

\section{Conclusion}

\frame{
\frametitle{Conclusion}

\begin{itemize}[<+->]
    \item Verbal reduplication in Mandarin Chinese is handled by a lexical rule.
    \item Different forms of reduplication are captured in a type hierarchy using underspecified lists.
    \item The interaction between reduplication and aspect marking is handled by multiple inheritance.
    \item Aspect marking is implemeted, interaction with reduplication will be implemented soon.
    
\end{itemize}
}
